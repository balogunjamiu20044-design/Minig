// Global Rate Limiter to prevent brute force attacks const apiLimiter = rateLimit({ windowMs: 15 * 60 * 1000, // 15 minutes max: 100, // Limit each IP to 100 requests per window message: { error: 'Too many requests from this IP, please try again later.' } });
